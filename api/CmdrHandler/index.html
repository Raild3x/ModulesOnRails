CmdrHandler

Package entrypoint file. Use this to initialize the client and server files.

local CmdrHandler = require(script.CmdrHandler)
local CmdrServer = CmdrHandler.Server

CmdrServer:Init()

-- Client
local CmdrHandler = require(script.CmdrHandler)
local CmdrClient = CmdrHandler.Client

CmdrClient:Init()
